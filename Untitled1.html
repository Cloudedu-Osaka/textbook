# (https://en.wikipedia.org/wiki/Methods_of_computing_square_roots#Decimal_(base_10))

import itertools

def long_division(s):
    # split by two digits
    si, sf = str(s).split('.')
    si = '0'*(len(si)%2) + si
    sf = sf + '0'*(len(sf)%2)
    
    sl = itertools.chain(*[[sx[i:i+2] for i in range(0,len(sx),2)]
                           for sx in (si, sf)])
    
    # find x, y and p
    p, r = 0, 0
    for c in map(int, sl):
        c += r*100
        x = 0
        while (2*(10*p)+(x+1))*(x+1) <= c:
            x+=1
        y = (20*p+x)*x
        p = 10*p+x     # x is 'quotient'
        r = c - y      # r is 'remainder'
    return p, r
